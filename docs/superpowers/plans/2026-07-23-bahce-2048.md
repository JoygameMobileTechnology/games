> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

```js
const SIZE = 4;

function emptyBoard() { return Array.from({ length: SIZE }, () => Array(SIZE).fill(0)); }

// Bir satırı sola kaydırır. merges: [{value, idx}] — idx: birleşmenin sonuç satırındaki konumu
function slideRow(row) {
  const vals = row.filter(v => v);
  const out = [], merges = []; let gained = 0;
  for (let i = 0; i < vals.length; i++) {
    if (vals[i] === vals[i + 1]) {
      const nv = vals[i] * 2;
      out.push(nv); merges.push({ value: nv, idx: out.length - 1 });
      gained += nv; i++;
    } else out.push(vals[i]);
  }
  while (out.length < SIZE) out.push(0);
  return { row: out, gained, merges };
}

// dir: 'left' | 'right' | 'up' | 'down'
function move(board, dir) {
  const b = board.map(r => r.slice());
  const get = {
    left:  i => b[i].slice(),
    right: i => b[i].slice().reverse(),
    up:    i => b.map(r => r[i]),
    down:  i => b.map(r => r[i]).reverse(),
  }[dir];
  const set = {
    left:  (i, row) => { b[i] = row; },
    right: (i, row) => { b[i] = row.slice().reverse(); },
    up:    (i, row) => { row.forEach((v, j) => b[j][i] = v); },
    down:  (i, row) => { row.slice().reverse().forEach((v, j) => b[j][i] = v); },
  }[dir];
  const toYX = {
    left:  (i, j) => [i, j],
    right: (i, j) => [i, SIZE - 1 - j],
    up:    (i, j) => [j, i],
    down:  (i, j) => [SIZE - 1 - j, i],
  }[dir];
  let gained = 0, moved = false; const merges = [];
  for (let i = 0; i < SIZE; i++) {
    const line = get(i);
    const res = slideRow(line);
    if (res.row.join() !== line.join()) moved = true;
    gained += res.gained;
    for (const m of res.merges) { const [y, x] = toYX(i, m.idx); merges.push({ value: m.value, y, x }); }
    set(i, res.row);
  }
  return { board: b, gained, merges, moved };
}

function spawn(board, rnd) {
  const empties = [];
  board.forEach((r, y) => r.forEach((v, x) => { if (!v) empties.push([y, x]); }));
  if (!empties.length) return board;
  const [y, x] = empties[Math.floor(rnd() * empties.length)];
  const b = board.map(r => r.slice());
  b[y][x] = rnd() < 0.9 ? 2 : 4;
  return b;
}

function canMove(board) {
  for (let y = 0; y < SIZE; y++) for (let x = 0; x < SIZE; x++) {
    const v = board[y][x];
    if (!v) return true;
    if (x + 1 < SIZE && board[y][x + 1] === v) return true;
    if (y + 1 < SIZE && board[y + 1][x] === v) return true;
  }
  return false;
}

const CORE = { SIZE, emptyBoard, slideRow, move, spawn, canMove };
```

function applyResource(g, type, rnd) {
  const events = [];
  if (type === 'cicek_ozu') {
    const filled = [];
    for (let z = 0; z < g.unlocked; z++)
      ZONES[z].slots.forEach((s, i) => { if (s.type === 'cicek' && g.plots[z + ':' + i]) filled.push(z + ':' + i); });
    if (!filled.length) { g.bank[type] = (g.bank[type] || 0) + 1; return [{ kind: 'bank', type }]; }
    const key = filled[Math.floor(rnd() * filled.length)];
    g.variants[key] = (g.variants[key] || 0) + 1;
    return [{ kind: 'variant', key }];
  }
  const want = SLOT_FOR[type];
  const z = activeZone(g);
  const idx = ZONES[z].slots.findIndex((s, i) => s.type === want && !g.plots[z + ':' + i]);
  if (idx >= 0) {
    g.plots[z + ':' + idx] = { stage: 1 };
    events.push({ kind: 'plant', zone: z, slot: idx, type });
    if (zoneComplete(g, z) && g.unlocked < ZONES.length) {
      g.unlocked++;
      events.push({ kind: 'unlock', zone: g.unlocked - 1 });
      events.push(...drainBank(g, rnd));
    }
    return events;
  }

    g.animals[z]++;

  const events = [];

      events.push(...evs);
    }
  }
  return events;
}

const IDLE_STEP_MS = 4 * 3600 * 1000;

function idleGrowth(g, elapsedMs) {
  const steps = Math.min(2, Math.floor(elapsedMs / IDLE_STEP_MS));
  let grown = 0;
  if (steps > 0) for (const k in g.plots) {

    const ns = Math.min(MAX_STAGE, p.stage + steps);
    if (ns > p.stage) { p.stage = ns; grown++; }

}
